    <script>

        const urlParams = new URLSearchParams(window.location.search);
        const playerId = urlParams.get('id');

        async function loadProfile() {
            if (!playerId) {
                showToast('HATA: Oyuncu ID bulunamadı.', 'error');
                return;
            }
            try {
                const response = await fetch(`/api/player/profile?id=${playerId}`);
                const data = await response.json();

                if (data.success) {
                    const p = data.profile;
                    document.getElementById('profile-content').style.display = 'grid';
                    document.getElementById('prof-username').innerText = p.username;
                    document.getElementById('card-username').innerText = p.username;
                    document.getElementById('prof-accid').innerText = `@${p.accountId}`;
                    document.getElementById('prof-club').innerText = p.clubName;
                    document.getElementById('prof-ip').innerText = p.ip;
                    document.getElementById('prof-device').innerText = p.device;
                    document.getElementById('prof-lastseen').innerText = p.lastLogin;

                    document.getElementById('stat-level').innerText = p.level;
                    document.getElementById('stat-trophies').innerText = p.trophies;
                    document.getElementById('stat-gems').innerText = p.gems;
                    document.getElementById('stat-coins').innerText = p.coins;
                    document.getElementById('stat-friends').innerText = p.friendCount;
                    document.getElementById('stat-tickets').innerText = p.ticketCount;
                    document.getElementById('stat-quests').innerText = p.questCount;
                    document.getElementById('stat-visuals').innerText = `ID ${p.avatarId} | C ${p.nameColorId}`;
                    document.getElementById('prof-email').innerText = p.email;
                    document.getElementById('stat-chatban').innerText = p.chatBan ? 'YES AKLI' : 'TEMİZ';
                    document.getElementById('stat-chatban').style.color = p.chatBan ? '#ef4444' : '#10b981';
                    document.getElementById('stat-ticketban').innerText = p.ticketBan ? 'YASAKLI' : 'TEMİZ';
                    document.getElementById('stat-ticketban').style.color = p.ticketBan ? '#ef4444' : '#10b981';

                    // Badges
                    const rolesContainer = document.getElementById('prof-roles');
                    rolesContainer.innerHTML = '';
                    p.roles.forEach(role => {
                        const b = document.createElement('span');
                        b.className = 'role-badge';
                        b.innerText = role;
                        rolesContainer.appendChild(b);
                    });

                    if (p.premium) {
                        document.getElementById('card-premium').innerHTML = `<span class="premium-badge"><i class="fas fa-crown"></i> PREMIUM ÜYE</span><div style="font-size:0.7rem; color:var(--text-secondary); margin-top:0.3rem">${p.premiumEndTime}</div>`;
                    }

                    // Status
                    const statusBadge = document.getElementById('prof-status-badge');
                    if (p.status === 'Online') {
                        statusBadge.innerHTML = `<span class="status-badge status-online" style="padding: 0.75rem 1.5rem; font-size: 1rem;">● ONLINE</span>`;
                    } else {
                        statusBadge.innerHTML = `<span class="status-badge status-offline" style="padding: 0.75rem 1.5rem; font-size: 1rem;">○ OFFLINE</span>`;
                    }

                    // Ban
                    const banBox = document.getElementById('ban-info-box');
                    const banText = document.getElementById('ban-status-text');
                    const banReason = document.getElementById('ban-reason-text');
                    const banAction = document.getElementById('ban-action-btn');

                    banAction.innerHTML = '';
                    const kickBtn = document.createElement('button');
                    kickBtn.className = 'kick-btn';
                    kickBtn.innerHTML = '<i class="fas fa-door-open"></i> KOV';
                    kickBtn.onclick = () => kickPlayer(p.accountId);
                    banAction.appendChild(kickBtn);

                    if (p.banStatus === 'Yasaklı') {
                        banBox.style.borderColor = '#ef4444';
                        banBox.style.background = 'rgba(239, 68, 68, 0.05)';
                        banText.innerText = 'HESAP YASAKLANMIŞ';
                        banText.style.color = '#ef4444';
                        banReason.innerText = `Neden: ${p.banReason}`;

                        const unbanBtn = document.createElement('button');
                        unbanBtn.className = 'unban-btn';
                        unbanBtn.innerHTML = '<i class="fas fa-unlock"></i> YASAĞI KALDIR';
                        unbanBtn.onclick = () => unbanPlayer(p.accountId);
                        banAction.appendChild(unbanBtn);
                    } else {
                        banText.innerText = 'TEMİZ HESAP';
                        banText.style.color = '#10b981';
                        banReason.innerText = 'Hesaba ait aktif bir yasaklama bulunmuyor.';

                        const banBtn = document.createElement('button');
                        banBtn.className = 'ban-btn';
                        banBtn.innerHTML = '<i class="fas fa-ban"></i> YASAKLA';
                        banBtn.onclick = () => banPlayer(p.accountId);
                        banAction.appendChild(banBtn);
                    }
                } else {
                    showToast('Hata: ' + data.message, 'error');
                }
            } catch (e) { showToast('Profil fetch failed: ' + e.message, 'error'); }
        }

        async function unbanPlayer(id) {
            Modal.confirm('Yasağı Kaldır', 'Bu oyuncunun yasaklamasını kaldırmak istediğinize emin misiniz?', async () => {
                try {
                    const res = await fetch('/api/player/unban', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ accountId: id })
                    });
                    const result = await res.json();
                    showToast(result.message);
                    loadProfile();
                } catch (e) { showToast('Hata: ' + e.message, 'error'); }
            });
        }

        async function banPlayer(id) {
            Modal.prompt('Oyuncuyu Yasakla', 'Bu oyuncuyu yasaklamak için bir neden belirtin:', 'Kural İhlali', async (reason) => {
                if (!reason) return;
                try {
                    const res = await fetch('/api/player/ban', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ accountId: id, reason: reason })
                    });
                    const result = await res.json();
                    showToast(result.message, result.success ? 'success' : 'error');
                    loadProfile();
                } catch (e) { showToast('Hata: ' + e.message, 'error'); }
            });
        }

        async function kickPlayer(id) {
            Modal.confirm('Oyuncuyu Kov', 'Bu oyuncuyu sunucudan atmak istediğinize emin misiniz?', async () => {
                try {
                    const res = await fetch('/api/player/kick', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ accountId: id })
                    });
                    const result = await res.json();
                    showToast(result.message, result.success ? 'success' : 'error');
                    loadProfile();
                } catch (e) { showToast('Hata: ' + e.message, 'error'); }
            });
        }

        loadProfile();
    </script>
